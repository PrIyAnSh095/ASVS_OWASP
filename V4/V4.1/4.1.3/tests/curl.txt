# Secure lab curl examples
curl -i http://localhost:5000/ \
  -H "X-Forwarded-For: 127.0.0.1" \
  -H "X-Forwarded-Host: attacker.example.com" \
  -H "X-Forwarded-Proto: https" \
  -H "X-Real-IP: 10.0.0.1" \
  -H "X-User-ID: admin"
