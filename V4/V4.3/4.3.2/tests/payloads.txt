# Basic Introspection
query { __schema { types { name } } }

# Deep Introspection (Dumping Fields and Types)
query {
  __schema {
    types {
      name
      kind
      description
      fields {
        name
        type {
          name
          kind
          ofType {
            name
            kind
          }
        }
      }
    }
  }
}

# Type Introspection
query {
  __type(name: "User") {
    name
    fields {
      name
      type {
        name
      }
    }
  }
}
