# ASVS 4.2.3 curl Command Examples

## Health Check

curl -v http://localhost:8000/health
curl -v http://localhost:8001/health

## Valid Request (Both Should Accept)

curl -v -H "Accept: application/json" http://localhost:8000/api/test
curl -v -H "Accept: application/json" http://localhost:8001/api/test

## Test Transfer-Encoding

# Secure - should reject (400)
curl -v --http2 -H "Transfer-Encoding: chunked" http://localhost:8000/api/test

# Vulnerable - should accept (200)
curl -v --http2 -H "Transfer-Encoding: chunked" http://localhost:8001/api/test

## POST with JSON Body

# Secure - valid headers should pass (200)
curl -v --http2 \
  -X POST \
  -H "Content-Type: application/json" \
  -d '{"key": "value"}' \
  http://localhost:8000/api/test

# Secure - forbidden header should reject (400)
curl -v --http2 \
  -X POST \
  -H "Content-Type: application/json" \
  -H "Transfer-Encoding: chunked" \
  -d '{"key": "value"}' \
  http://localhost:8000/api/test

## Batch Test Script

#!/bin/bash

# Test each forbidden header
for header in "Transfer-Encoding: chunked" \
              "Connection: close" \
              "Keep-Alive: timeout=5" \
              "Upgrade: h2c" \
              "Proxy-Connection: keep-alive" \
              "Trailer: X-Signature"
do
  echo "Testing: $header"
  echo "=== SECURE (should be 400) ==="
  curl -s -w "Status: %{http_code}\n" -H "$header" http://localhost:8000/api/test | head -1
  
  echo "=== VULNERABLE (should be 200) ==="
  curl -s -w "Status: %{http_code}\n" -H "$header" http://localhost:8001/api/test | head -1
  echo ""
done

## Test Case Matrix (Automated)

#!/bin/bash

SECURE="http://localhost:8000"
VULNERABLE="http://localhost:8001"

headers=(
  "Transfer-Encoding:chunked"
  "Connection:close"
  "Keep-Alive:timeout=5"
  "Upgrade:h2c"
  "Proxy-Connection:keep-alive"
  "Trailer:X-Signature"
)

printf "%-25s | %-10s | %-10s\n" "Header" "Secure (400)" "Vulnerable (200)"
echo "=================================================="

for header in "${headers[@]}"; do
  secure_status=$(curl -s -w "%{http_code}" -o /dev/null -H "$header" "$SECURE/api/test")
  vuln_status=$(curl -s -w "%{http_code}" -o /dev/null -H "$header" "$VULNERABLE/api/test")
  
  secure_pass="✓" && [ "$secure_status" != "400" ] && secure_pass="✗"
  vuln_fail="✗" && [ "$vuln_status" == "200" ] && vuln_fail="✓"
  
  printf "%-25s | %s %3s | %s %3s\n" "$header" "$secure_pass" "$secure_status" "$vuln_fail" "$vuln_status"
done

## Notes

# - Use --http2 flag for HTTP/2 support (requires curl with h2 support)
# - Check curl version: curl --version | grep h2
# - Some systems may need: apt-get install curl-http2 (or equivalent)
# - Use -v flag for verbose output including headers
# - Use jq for pretty JSON formatting: curl -s ... | jq .
# - Use -w flag for timing and status code information
